 6 ; When opening page, load gschemrc file from the page's directory

 8 
 9 
10 
11 ( use-modules ( lepton rc      ) )
12 ( use-modules ( lepton page    ) )
13 ( use-modules ( lepton log     ) )
14 ( use-modules ( schematic hook ) )
15 
16 
17 ( define ( load-gschemrc-file page )
18     ( chdir ( dirname (page-filename page) ) )
19     ( parse-rc "lepton-schematic" "gschemrc" )
20     ( log! 'warning "load-gschemrc.scm: Loaded RC file [~a/gschemrc]"   (getcwd) )
21     ( format #t     "load-gschemrc.scm: Loaded RC file [~a/gschemrc]~%" (getcwd) )
22 )
23 
24 
25 ( define ( install_hooks )
26     ( add-hook! open-page-hook load-gschemrc-file #f ) ; #f => prepend
27 )
28 
29 
30 ( install_hooks )
