 5 ; this Guile Scheme script prints current lepton-schematic
 6 ;   key bindings to stdout
 7 ; usage: in lepton-schematic ":" prompt or in File->REPL type:
 8 ;   ( primitive-load "/path/to/list-keys-dmn.scm" )
 9 ;
10 ; vim: ft=scheme tabstop=2 softtabstop=2 shiftwidth=2 expandtab
11 ;
12 
13 ( use-modules ( schematic gui keymap ) )
14 ( use-modules ( gschem keymap ) )
15 ( use-modules ( gschem action ) )
16 
17 
18 ; function: print-keymap()
19 ;
20 ( define* ( print-keymap kmap #:optional (indent 0) )
21 
22   ; auxiliary local functions:
23 
24   ( define ( print-indent ind )
25     ( while ( > ind 0 )
26       ( format #t "  " )
27       ( set! ind (1- ind) )
28     )
29   )
30 
31   ( define ( print-key key )
32     ( format #t "~a" (key->display-string key) )
33   )
34 
35   ( define ( print-action action )
36   ( let*
37     (
38     ( action-str (format #f "~a" action) )
39     ( action-obj (eval-string action-str (current-module)) )
40     )
41 
42     ( format #t "~20t= ~a" action )
43     ( if ( action? action-obj )
44       ( format #t " (~a)" ( action-property action-obj 'label ) )
45     )
46     ( format #t "~%" )
47   )
48   )
49 
50 
51   ; function body:
52 
53   ( format #t "~%" )
54 
55   ( keymap-for-each
56     ( lambda( key action-or-keymap )
57 
58       ( print-indent indent )
59       ( print-key key )
60 
61       ( if ( keymap? action-or-keymap )
62         ( print-keymap action-or-keymap (1+ indent) ) ; if
63         ( print-action action-or-keymap )             ; else
64       )
65 
66     )
67     kmap
68   )
69 
70 ) ; print-keymap()
71 
72 
73 ; top-level code:
74 ;
75 ( print-keymap %global-keymap )
