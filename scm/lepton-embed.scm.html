 30 ; command line options:
 31 ;
 32 ( define cmd-line-args-spec
 33 ( list
 34   ( list ; --embed (-e)
 35     'embed
 36     ( list 'single-char #\e )
 37     ( list 'value        #f )
 38   )
 39   ( list ; --unembed (-u)
 40     'unembed
 41     ( list 'single-char #\u )
 42     ( list 'value        #f )
 43   )
 44   ( list ; --help (-h)
 45     'help
 46     ( list 'single-char #\h )
 47     ( list 'value        #f )
 48   )
 49   ( list ; --version (-V)
 50     'version
 51     ( list 'single-char #\V )
 52     ( list 'value        #f )
 53   )
 54 )
 55 ) ; cmd-line-args-spec

 59 ( define ( usage exit-code )

 62 
 63 Lepton EDA component embedding/unembedding utility.
 64 
 65 Options:

 83 ( define ( version )
 84   ( format #t "~a~%" ( lepton-version 'msg ) )
 85   ( primitive-exit 0 )

 89 
 90 ( define ( page-open file )
 91   ( catch #t
 92   ( lambda()
 93     ; return:
 94     ( file->page file )
 95   )
 96   ( lambda( ex . args )
 97     ( format (current-error-port)
 98               "Cannot open file [~a]:~%  '~a: ~a ~%"
 99               file ex args )
100     ; return:
101     #f
102   )
103   ) ; catch
104 ) ; page-open()
105 
106 
107 
108 ( define* ( page-save page #:optional (suffix "") )
109 ( let
110   (
111   ( file ( page-filename page ) )
112   )
113 
114   ( catch #t
115   ( lambda()
116     ( with-output-to-file
117       ( format #f "~a~a" file suffix )
118       ( lambda()
119         ; return:
120         ( format #t "~a" (page->string page) )
121       )
122     )
123   )
124   ( lambda( ex . args )
125     ( format (current-error-port)
126               "Cannot save file [~a]:~%  '~a: ~a ~%"
127               file ex args )
128     ; return:
129     #f
130   )
131   ) ; catch
132 
133 ) ; let
134 ) ; page-save()
135 
136 
137 
138 ( define ( do-embed page embed )
139 
140   ( define ( embeddable? obj )
141     ( or (component? obj) (picture? obj) )
142   )
143 
144   ( for-each
145   ( lambda( comp )
146       ( set-object-embedded! comp embed )
147   )
148   ( filter embeddable? (page-contents page) )
149   )
150 
151 ) ; do-embed()
152 
153 
154 
155 ( define ( process-file file embed )
156 ( let
157   (
158   ( page ( page-open file ) )
159   )
160 
161   ( when page
162     ( do-embed page embed )
163     ( if ( and (page-dirty? page) (page-save page) )
164       ( format #t "Saved: [~a]~%" file )
165     )
166   )
167 
168 ) ; let
169 ) ; process-file()
170 
171 
172 
173 ( define ( main )
174 ( let
175   (
176   ( cmd-line-args '() )
177   ( files         '() )
178   ( arg-embed      #f )
179   ( arg-unembed    #f )
180   )
181 
182   ( set! cmd-line-args
183     ( getopt-long (program-arguments) cmd-line-args-spec )
184   )
185 
186   ( if ( option-ref cmd-line-args 'help #f )
187     ( usage 0 )
188   )
189   ( if ( option-ref cmd-line-args 'version #f )
190     ( version )
191   )
192 
193   ( set! arg-embed   ( option-ref cmd-line-args 'embed #f ) )
194   ( set! arg-unembed ( option-ref cmd-line-args 'unembed #f ) )
195 
196   ( if ( and arg-embed arg-unembed )
197     ( usage 1 )
198   )
199   ( if ( and (not arg-embed) (not arg-unembed) )
200     ( usage 2 )
201   )
202 
203   ( set! files ( option-ref cmd-line-args '() '() ) )
204   ( if ( null? files )
205     ( usage 3 )
206   )
207 
208 
209   ( (@@ (guile-user) parse-rc) "lepton-embed" "gafrc" )
210 
211   ( for-each
212   ( lambda( file )
213     ( process-file file arg-embed )
214   )
215     files
216   )
217 
218 ) ; let
219 ) ; main()
220

227     ( main )
