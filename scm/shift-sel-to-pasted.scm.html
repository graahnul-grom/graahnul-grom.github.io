 9 ;   ( primitive-load "/path/to/shift-sel-to-pasted.scm" )
10 ;
11 
12 
13 ( use-modules ( gschem selection ) )
14 ( use-modules ( gschem window    ) )
15 ( use-modules ( gschem hook      ) )
16 
17 
18 ( define ( select-pasted objs )
19 
20   ( for-each deselect-object! (page-selection (active-page)) )
21   ( for-each select-object! objs )
22 
23 ) ; select-pasted()
24 
25 
26 ; top-level code:
27 ;
28 ( add-hook! paste-objects-hook select-pasted )
