  7 ; Select locked components with keyboard shortcuts:
  8 ;   - select all locked components
  9 ;   - select next locked component
 10 ;   - select locked component under mouse cursor
 11 ;
 12 ; Usage:
 13 ; 1) customize the shortcuts defined below if needed (for examples
 14 ;    see $PREFIX/share/lepton-eda/scheme/conf/schematic/keys.scm),
 15 ;    or comment out to disable
 16 ; 2) type in the ":" prompt or add it to your gschemrc file:
 17 ;    ( primitive-load "/path/to/select-locked.scm" )
 18 ;
 19 ( define kbind-select-all          "<Control><Shift>L" )
 20 ( define kbind-select-next         "<Shift>L" )
 21 ( define kbind-select-under-mouse  "<Control>L" )
 22 ( define verbose #t ) ; replace "#t" with "#f" if needed

 50 ( define ( dbg-mk-box-around-comp comp ) ; [DEBUG]
 51   ( define ( non-txt-obj? o )
 52     ( not (text? o) )
 53   )

 63   )

 95 ( define ( select-comp comp )
 96   ( when comp
 97     ( select-object! comp )
 98     ( for-each
 99       select-object!
100       ( object-attribs comp )
101     )
102   )

108 ; public:
109 ;
110 ( define ( select-next-locked-comp )
111 ( let*
112   (
113   ( comps ( locked-comps ) )
114   ( len   ( length comps ) )
115   ( sel   ( page-selection (active-page) ) )
116   ( ndx   0 )
117   ( comp #f )
118   )
119 
120   ; helper functions:
121 
122   ( define ( print-msg i c ) ; i: index, c: component
123     ( log! 'message
124       "select-locked: ~a of ~a (~a)~a"
125       ( 1+ i )
126       len
127       ( component-basename c )
128       ( if verbose
129         ". press <E E> to edit, <E Shift+L> to unlock" ; if
130         ""                                             ; else
131       )
132     )
133   )
134 
135   ( define ( next-ndx i )
136     ; return:
137     ( if ( = i (1- len) )
138       0        ; if
139       ( 1+ i ) ; else
140     )
141   )
142 
143   ( define ( go? c )
144     ; return:
145     ( or
146       ( null? sel )                  ; nothing selected
147       ( not (find locked-obj? sel) ) ; no locked comps selected
148       ( object-selected? c )         ; locked comp c selected
149     )
150   )
151 
152   ; function's body:
153 
154   ( when ( null? comps )
155     ( log! 'message "select-locked: no locked components found" )
156   )
157 
158   ( while ( < ndx len )
159     ( set! comp ( list-ref comps ndx ) )
160 
161     ( when ( go? comp )
162       ( deselect-all )
163       ( set! comp (list-ref comps (next-ndx ndx) ) )
164       ( select-comp comp )
165       ( print-msg ndx comp )
166       ( break )
167     )
168 
169     ( set! ndx (1+ ndx) )
170   )
171 
172 ) ; let
173 ) ; select-next-locked-comp()

178 ; public:
179 ;
180 ( define ( select-all-locked-comps )
181 ( let
182   (
183   ( comps ( locked-comps ) )
184   )
185 
186   ( unless ( null? comps )
187     ( deselect-all )
188     ( for-each select-comp comps )
189   )
190 
191   ( if ( null? comps )
192     ( log! ; if
193       'message
194       "select-locked: no locked components found"
195     )
196     ( log! ; else
197       'message
198       "select-locked: ~a locked component(s) selected~a"
199       ( length comps )
200       ( if verbose ". press <E Shift+L> to unlock" "" )
201     )
202   )
203 
204 
205 ) ; let
206 ) ; select-all-locked-comps()

211 ; public:
212 ;
213 ( define ( select-locked-comp-under-mouse )
214 
215   ; helper functions:
216 
217   ( define ( point-in-bounds? pt bounds )
218   ( let*
219     (
220     ( top-left  ( car bounds ) )
221     ( bot-right ( cdr bounds ) )
222     ( x1  ( car top-left ) )
223     ( x2  ( car bot-right ) )
224     ( y1  ( cdr bot-right ) )
225     ( y2  ( cdr top-left ) )
226     ( ptx ( car pt ) )
227     ( pty ( cdr pt ) )
228     )
229 
230     ; return:
231     ( and
232       ( >= ptx x1 ) ( <= ptx x2 )
233       ( >= pty y1 ) ( <= pty y2 )
234     )
235 
236   ) ; let
237   ) ; point-in-bounds?()
238 
239 
240   ( define ( comp-bounds c ) ; comp's bounds w/o attrs
241     ( define ( non-txt-obj? o )
242         ( not (text? o) )
243     )
244   ( let*
245     (
246     ( cc ( component-contents c ) )
247     ( oo ( filter non-txt-obj? cc ) )
248     ( bb ( map object-bounds oo ) )
249     )
250 
251     ; return:
252     ( apply fold-bounds bb )
253 
254   ) ; let
255   ) ; comp-bounds()
256 
257 
258   ( define ( comp-area c )
259   ( let*
260     (
261     ( bounds    ( comp-bounds c ) )
262     ( top-left  ( car bounds ) )
263     ( bot-right ( cdr bounds ) )
264     ( x1 ( car top-left ) )
265     ( x2 ( car bot-right ) )
266     ( y1 ( cdr bot-right ) )
267     ( y2 ( cdr top-left ) )
268     )
269 
270     ; return:
271     ( * (- x2 x1) (- y2 y1) )
272 
273   ) ; let
274   ) ; comp-area()
275 
276 
277   ( define ( less-area? c1 c2 )
278     ( < (comp-area c1) (comp-area c2) )
279   )
280 
281 
282   ( define ( comp-under-mouse? c )
283     ; return:
284     ( if ( and (pointer-position) (comp-bounds c) )
285       ( point-in-bounds? (pointer-position) (comp-bounds c) ) ; if
286       #f                                                      ; else
287     )
288   )
289 
290   ; function's body:
291 
292 ( let*
293   (
294   ; sort comps => the smallest will be found first, not a title block:
295   ;
296   ( comps ( sort (locked-comps) less-area? ) )
297   ( comp  ( find comp-under-mouse? comps ) )
298   )
299 
300   ( deselect-all )
301 
302   ( if comp
303     ( select-comp comp ) ; if
304     ( log! 'message      ; else
305       "select-locked: no locked component under mouse cursor" )
306   )
307 
308 ) ; let
309 ) ; select-locked-comp-under-mouse()
